use crate::address::Address;
use ic::inet::{link, new_inet, new_node, port, INet, CON, ERA};
use num_traits::{PrimInt, Unsigned};
use serde::{Deserialize, Serialize};

#[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
pub enum Tree {
  N,
  B { lft: Box<Tree>, rgt: Box<Tree> },
}

impl Tree {
  pub fn new() -> Tree {
    Tree::N
  }

  // Converting a tree into a Hash
  pub fn to_bytes(&self) -> Vec<u8> {
    let mut byte_array = Vec::new();
    let mut current_byte = 0u8;
    let mut bits_written = 0u8;

    // Serialize the tree into bytes.
    self.to_bytes_internal(&mut byte_array, &mut current_byte, &mut bits_written);

    // If there are remaining bits that haven't been pushed to the vector, push them.
    if bits_written != 0 {
      byte_array.push(current_byte);
    }

    byte_array
  }

  pub fn to_bytes_internal(&self, byte_array: &mut Vec<u8>, current_byte: &mut u8, bits_written: &mut u8) {
    *current_byte <<= 1;
    *bits_written += 1;

    match self {
      Tree::N => {}
      Tree::B { lft, rgt } => {
        *current_byte += 1;

        // Serialize the left and right subtrees.
        lft.to_bytes_internal(byte_array, current_byte, bits_written);
        rgt.to_bytes_internal(byte_array, current_byte, bits_written);
      }
    }

    // If we've written a full byte, push it to the vector and reset our counters.
    if *bits_written == 8 {
      byte_array.push(*current_byte);
      *current_byte = 0;
      *bits_written = 0;
    }
  }

  // Example of using to_vec with to_u32
  // https://i.imgur.com/rSTojwV.png
  pub fn to_inet_as_program(&self) -> INet {
    let nodes = Tree::to_vec_as_balanced_tree(&self, Tree::to_unsigned::<u32>);
    INet { nodes, reuse: vec![], rules: 0 }
  }

  // How the tree is converted into a input for the contract
  // https://i.imgur.com/7EVIHgN.png
  pub fn to_inet_as_input(&self) -> INet {
    let mut inet = new_inet();
    self.to_inet_as_input_internal(&mut inet, port(0, 1));
    inet
  }
  pub fn to_inet_as_input_internal(&self, inet: &mut INet, port_parent: u32) {
    match self {
      Tree::N {} => {
        let lam_b = new_node(inet, CON);
        let era_n = new_node(inet, ERA);
        let lam_n = new_node(inet, CON);
        link(inet, port(lam_b, 0), port_parent);
        link(inet, port(lam_b, 1), port(era_n, 0));
        link(inet, port(lam_b, 2), port(lam_n, 0));
        link(inet, port(era_n, 1), port(era_n, 2));
        link(inet, port(lam_n, 1), port(lam_n, 2));
      }
      Tree::B { lft, rgt } => {
        let lam_b = new_node(inet, CON);
        let lam_n = new_node(inet, CON);
        let era_n = new_node(inet, ERA);
        let app_0 = new_node(inet, CON);
        let app_1 = new_node(inet, CON);
        link(inet, port(lam_b, 0), port_parent);
        link(inet, port(lam_b, 1), port(app_1, 0));
        link(inet, port(lam_b, 2), port(lam_n, 0));
        link(inet, port(lam_n, 1), port(era_n, 0));
        link(inet, port(lam_n, 2), port(app_0, 2));
        link(inet, port(era_n, 1), port(era_n, 2));
        link(inet, port(app_0, 0), port(app_1, 2));
        lft.to_inet_as_input_internal(inet, port(app_1, 1));
        rgt.to_inet_as_input_internal(inet, port(app_0, 1));
      }
    }
  }

  /// Converts any primitive unsigned type into a Tree.
  /// The Tree is encoded as a bit stream where taking a
  /// left branch means a '0' bit and a right branch a '1' bit.
  /// Ex: 0b0 = (B (B L L) L). 0b11 (B L (B L (B L L))).
  pub fn from_unsigned<T: Unsigned + PrimInt>(bits: T) -> Self {
    if bits == T::zero() {
      let tree = Tree::B {
        lft: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
        rgt: Box::new(Tree::N),
      };
      return tree;
    }
    Tree::from_unsigned_internal(bits)
  }

  /// Internal recursive call for `from_unsigned`.
  fn from_unsigned_internal<T: Unsigned + PrimInt>(bits: T) -> Self {
    if bits == T::zero() {
      let tree = Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) };
      return tree;
    }
    let next = bits >> 1;
    if (bits & T::one()) == T::zero() {
      // Go left
      Tree::B { lft: Box::new(Tree::from_unsigned_internal(next)), rgt: Box::new(Tree::N) }
    } else {
      // Go right
      Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::from_unsigned_internal(next)) }
    }
  }

  /// Converts a bit tree into a primitive unsigned integer.
  /// Ignores any potentially malformed parts of the tree
  pub fn to_unsigned<T: Unsigned + PrimInt>(&self) -> T {
    match self {
      Tree::N => T::zero(),
      Tree::B { lft, rgt } => match (lft.as_ref(), rgt.as_ref()) {
        (Tree::B { .. }, Tree::N) => lft.to_unsigned::<T>() << 1,
        (Tree::N, Tree::B { .. }) => T::one() | rgt.to_unsigned::<T>() << 1,
        _ => T::zero(),
      },
    }
  }

  /// Converts a slice into a balanced Tree by repeatedly splitting the slice in the middle.
  /// All the slice values are in the leaves of the balanced tree.
  pub fn from_slice_as_balanced_tree<A>(slice: &[A], convert: fn(&A) -> Self) -> Self {
    match slice {
      [] => Tree::N,
      [head] => convert(head),
      slice => {
        let (lft, rgt) = slice.split_at(slice.len() / 2);
        Tree::B {
          lft: Tree::from_slice_as_balanced_tree(lft, convert).into(),
          rgt: Tree::from_slice_as_balanced_tree(rgt, convert).into(),
        }
      }
    }
  }

  /// Converts a balanced tree into a vec
  pub fn to_vec_as_balanced_tree<A>(&self, convert: fn(&Self) -> A) -> Vec<A> {
    match self {
      Tree::N => {
        // TODO: This looks broken
        vec![convert(self)]
      }
      Tree::B { lft, rgt } => match (&**lft, &**rgt) {
        (Tree::B { .. }, Tree::B { .. }) => {
          let mut vec = lft.to_vec_as_balanced_tree(convert);
          vec.extend(rgt.to_vec_as_balanced_tree(convert));
          vec
        }
        _ => {
          let elem = vec![convert(self)];
          elem
        }
      },
    }
  }

  /// Converts a slice into a cons list encoded in the Tree
  /// [head tail] -> B { lft: head, rgt: tail}
  pub fn from_slice_as_list<A>(slice: &[A], convert: fn(&A) -> Self) -> Self {
    match slice.split_first() {
      Some((head, tail)) => {
        Tree::B { lft: convert(head).into(), rgt: Self::from_slice_as_list(tail, convert).into() }
      }
      None => Tree::N,
    }
  }

  /// Converts a Tree into a vec by interpreting it as a cons list
  /// B { lft: head, rgt: tail } -> [head tail]
  /// N => []
  pub fn to_vec_as_list<A>(&self, convert: fn(&Self) -> A) -> Vec<A> {
    fn go<A>(tree: &Tree, convert: fn(&Tree) -> A, xs: &mut Vec<A>) {
      match tree {
        Tree::N => {}
        Tree::B { lft, rgt } => {
          xs.push(convert(lft));
          go(&rgt, convert, xs);
        }
      }
    }
    let mut xs = vec![];
    go(self, convert, &mut xs);
    xs
  }

  pub fn from_inet(inet: &INet) -> Self {
    Tree::from_slice_as_balanced_tree(&inet.nodes, |x: &u32| (*x).into())
  }

  /// Converts a kindelia network account address into a Tree as if it were a u160
  pub fn from_address(addr: &Address) -> Self {
    Self::from_byte_array(addr.as_ref())
  }

  /// Convert a BE byte array into a bit tree (meaning byte 0 is at the end of tree)
  pub fn from_byte_array(arr: &[u8]) -> Self {
    Self::from_byte_array_internal(arr, Self::B { lft: Self::N.into(), rgt: Self::N.into() }.into())
  }

  fn from_byte_array_internal(arr: &[u8], mut tail: Box<Self>) -> Self {
    match arr.split_first() {
      Some((&byte, arr)) => {
        for i in 0 .. 8 {
          let bit = (byte >> (7 - i)) & 1; // Get i-th from msb to lsb
          if bit == 1 {
            // Add a right (1 bit)
            tail = Self::B { lft: Self::N.into(), rgt: tail }.into();
          } else {
            // Add a left (0 bit)
            tail = Self::B { lft: tail, rgt: Self::N.into() }.into();
          }
        }
        Self::from_byte_array_internal(arr, tail)
      }
      None => *tail,
    }
  }

  /// Converts a bit tree into a kindelia network account address.
  /// Returns `None` if the tree is not long enough or not a valid bit tree.
  pub fn try_to_address(&self) -> Option<Address> {
    let mut tree = self;
    let mut addr = [0; 20];
    for i in 0 .. 20 {
      for j in 0 .. 8 {
        match tree {
          Tree::B { lft, rgt } => match (lft.as_ref(), rgt.as_ref()) {
            // Going right
            (Tree::N, tree_ @ Tree::B { .. }) => {
              tree = tree_;
              addr[19 - i] |= 1 << j;
            }
            // Going left
            (tree_ @ Tree::B { .. }, Tree::N) => {
              tree = tree_;
              // No need to clear bit here as already initialized to zero.
            }
            // Unexpected end of tree
            (Tree::N, Tree::N) => return None,
            // Malformed bit tree
            (Tree::B { .. }, Tree::B { .. }) => return None,
          },
          Tree::N => return None,
        }
      }
    }
    if tree == &(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }) {
      // Correct ending
      Some(addr.into())
    } else {
      // Has more stuff after
      None
    }
  }

  /// Converts the tree to a pair of two values
  pub fn try_into_pair<L, R>(self, convert_fst: fn(Self) -> L, convert_snd: fn(Self) -> R) -> Option<(L, R)> {
    match self {
      Tree::N => None,
      Tree::B { lft, rgt } => Some((convert_fst(*lft), convert_snd(*rgt))),
    }
  }

  pub fn from_pair<L, R>(value: (&L, &R), convert_fst: fn(&L) -> Self, convert_snd: fn(&R) -> Self) -> Self {
    Tree::B { lft: convert_fst(value.0).into(), rgt: convert_snd(value.1).into() }
  }
}

impl<T: Unsigned + PrimInt> From<T> for Tree {
  fn from(value: T) -> Self {
    Self::from_unsigned(value)
  }
}

#[cfg(test)]
mod tests {
  use super::*;
  use crate::address::Address;

  #[test]
  fn test_tree_new() {
    let tree = Tree::new();
    assert_eq!(tree, Tree::N);
  }

  #[test]
  fn test_tree_to_bytes() {
    let tree = Tree::B {
      lft: Box::new(Tree::B {
        lft: Box::new(Tree::B {
          lft: Box::new(Tree::N),
          rgt: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
        }),
        rgt: Box::new(Tree::N),
      }),
      rgt: Box::new(Tree::N),
    };

    let bytes = tree.to_bytes();
    assert_eq!(bytes, vec![232, 0]);
  }

  #[test]
  fn test_tree_from_u32() {
    let tree = Tree::from(1 as u32);
    assert_eq!(tree, Tree::B {
      lft: Box::new(Tree::N),
      rgt: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) })
    });
  }

  #[test]
  fn test_tree_to_u32() {
    let tree = Tree::B {
      lft: Box::new(Tree::N),
      rgt: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
    };
    let result = tree.to_unsigned::<u32>();
    assert_eq!(result, 1);
  }

  #[test]
  fn test_to_vec() {
    let tree = Tree::B {
      lft: Box::new(Tree::B {
        lft: Box::new(Tree::N),
        rgt: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
      }),
      rgt: Box::new(Tree::B {
        lft: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
        rgt: Box::new(Tree::N),
      }),
    };

    let vec = tree.to_vec_as_balanced_tree(Tree::to_unsigned::<u32>);
    assert_eq!(vec, vec![1, 0]);
  }

  #[test]
  fn test_from_vec() {
    let vec = vec![1, 0];
    let tree = Tree::from_slice_as_balanced_tree(&vec, |x: &u32| Tree::from(*x));

    assert_eq!(tree, Tree::B {
      lft: Box::new(Tree::B {
        lft: Box::new(Tree::N),
        rgt: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
      }),
      rgt: Box::new(Tree::B {
        lft: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
        rgt: Box::new(Tree::N),
      }),
    });
  }

  #[test]
  fn test_round_trip_to_vec_from_vec() {
    let tree = Tree::B {
      lft: Box::new(Tree::N),
      rgt: Box::new(Tree::B { lft: Box::new(Tree::N), rgt: Box::new(Tree::N) }),
    };

    let vec = tree.to_vec_as_balanced_tree(Tree::to_unsigned::<u32>);
    let round_trip_tree = Tree::from_slice_as_balanced_tree(&vec, |x| Tree::from(*x));

    assert_eq!(tree, round_trip_tree);
  }

  #[test]
  fn test_round_trip_from_vec_to_vec() {
    let vec = vec![1, 0];
    let tree = Tree::from_slice_as_balanced_tree(&vec, |x: &u32| Tree::from(*x));
    let round_trip_vec = tree.to_vec_as_balanced_tree(Tree::to_unsigned::<u32>);

    assert_eq!(vec, round_trip_vec);
  }

  #[test]
  fn test_tree_from_pair() {
    let num = 0 as u32;
    let tree = Tree::from_pair((&num, &num), |x| Tree::from(*x), |x| Tree::from(*x));
    let expected = Tree::B { lft: Tree::from(num).into(), rgt: Tree::from(num).into() };
    assert_eq!(tree, expected)
  }

  #[test]
  fn test_tree_to_pair() {
    let num = 438 as u32;
    let tree = Tree::B { lft: Tree::from(num).into(), rgt: Tree::from(num).into() };
    let pair = tree.try_into_pair(|x| Tree::to_unsigned::<u32>(&x), |x| Tree::to_unsigned::<u32>(&x));
    let expected = Some((num, num));
    assert_eq!(pair, expected)
  }

  #[test]
  fn test_tree_pair_conversion_inverse() {
    let num = 5 as u64;
    let tree = Tree::B { lft: Tree::from(num).into(), rgt: Tree::from(num).into() };
    let from_tree = tree.try_into_pair(|x| Tree::to_unsigned::<u64>(&x), |x| Tree::to_unsigned::<u64>(&x));
    assert_eq!(Some((num, num)), from_tree)
  }

  #[test]
  fn test_tree_from_address() {
    let addr = Address([1; 20]);
    let tree = Tree::from_address(&addr);

    let mut expected = Box::new(Tree::B { lft: Tree::N.into(), rgt: Tree::N.into() });
    for _ in 0 .. 20 {
      for j in 0 .. 8 {
        if j == 7 {
          expected = Box::new(Tree::B { lft: Box::new(Tree::N), rgt: expected });
        } else {
          expected = Box::new(Tree::B { lft: expected, rgt: Box::new(Tree::N) });
        }
      }
    }
    assert_eq!(tree, *expected)
  }

  #[test]
  fn test_tree_to_address() {
    let mut tree = Box::new(Tree::B { lft: Tree::N.into(), rgt: Tree::N.into() });
    for _ in 0 .. 20 {
      for j in 0 .. 8 {
        if j == 7 || j == 6 {
          tree = Box::new(Tree::B { lft: Box::new(Tree::N), rgt: tree });
        } else {
          tree = Box::new(Tree::B { lft: tree, rgt: Box::new(Tree::N) });
        }
      }
    }
    let address = tree.try_to_address();
    let expected = Some([3 as u8; 20].into());
    assert_eq!(address, expected)
  }

  #[test]
  fn test_tree_address_conversion_inverse() {
    let addr = [1; 20].into();
    let tree = Tree::from_address(&addr);
    let back = tree.try_to_address().expect("tree to address conversion failed");
    assert_eq!(addr, back)
  }
