use crate::global_state::GlobalState;

use super::{block::Block, ConsensusError};

#[derive(Debug, Clone)]
pub struct Chain {
  blocks: imbl::Vector<Block>,
  state: GlobalState,
}

impl Chain {
  pub fn new(genesis: Block) -> Chain {
    Chain { blocks: imbl::vector![genesis], state: GlobalState::new() }
  }

  pub fn new_from_blocks(blocks: imbl::Vector<Block>) -> Result<Chain, ConsensusError> {
    let genesis = blocks.front().ok_or(ConsensusError::NoGenesis)?.clone();
    let rest = blocks.into_iter().skip(1);
    Chain::new(genesis).batch_process_block(rest)
  }

  pub fn process_block(self, block: Block) -> Result<Chain, ConsensusError> {
    self.batch_process_block([block])
  }

  pub fn batch_process_block<BlockIter>(self, blocks: BlockIter) -> Result<Chain, ConsensusError>
  where
    BlockIter: IntoIterator<Item = Block>,
  {
    let mut new_chain = self.clone();
    for block in blocks {
      let previous = &new_chain.blocks.last().ok_or(ConsensusError::NoGenesis)?.header;
      block.validate_header(previous)?;

      // TODO: maybe we can batch transact all blocks at once
      new_chain.state = new_chain.state.transact(&block).map_err(ConsensusError::BlockValidationFailed)?;

      new_chain.blocks.push_back(block);
    }
    Ok(new_chain)
  }

  pub fn replace_with_longest_chain(
    &mut self,
    remote_blocks: imbl::Vector<Block>,
  ) -> Result<(), ConsensusError> {
    if remote_blocks.len() > self.blocks.len() {
      *self = Chain::new_from_blocks(remote_blocks)?;
    }
    Ok(())
  }
}

#[cfg(test)]
mod tests {
  use crate::{
    crypto::KeyPair,
    global_state::{Payload, SpecificPayload, Transaction},
    hash::Hashable,
  };

  use super::*;

  #[test]
  fn test_process_block() {
    let a = KeyPair::gen();
    let b = KeyPair::gen();
    let miner = KeyPair::gen();

    let mut chain = Chain::new(Block::genesis());
    chain.state.set_balance(a.pubk.into(), 100);
    let txs = vec![Transaction::new(
      Payload { amount: 20, fee: 5, specific: SpecificPayload::SendFunds { to: b.pubk.into() } },
      a.seck,
    )];
    let block = Block::mine(chain.blocks[0].header.hash().unwrap(), miner.pubk.into(), txs).unwrap();
    chain.process_block(block).unwrap();
  }
}
